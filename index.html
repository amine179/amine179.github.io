def showNet():
    global ax, fig
    global inps, outs, xn, yn, pop, week_pop
    
    fig, ax = plt.subplots(figsize=(17,8))
    
    
    R0 = 0.02
    V_out_I = 0.02
    
    d = float("{:.2f}".format(outs[0][0]*R0))
    c = float("{:.2f}".format(1-d))
    b = float("{:.2f}".format(outs[0][1]))
    a = float("{:.2f}".format(1-b))
    f = float("{:.2f}".format(outs[0][2]))
    e = float("{:.2f}".format(1-f))
    
    death_rate = float("{:.2f}".format(outs[0][3]* V_out_I)) 
    h_score = float("{:.2f}".format(outs[0][4]))
    e_score = float("{:.2f}".format(outs[0][5]))
    
    G = nx.DiGraph()

    # Calculate for dynamic label changes    

    l1, l2, l3, l4, l5 ='N', 'S', 'I', 'U', 'K'

    edgelist= [(l1,l1),(l1,l2),(l2,l3), (l3,l4), (l3,l5)]

    e_labels = ['a: '+str(a), 'b: '+str(b), 'c: '+str(c), 'd: '+str(d), 'e: '+str(e), 'f: '+str(f)] 

#     e_labels = ['a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k/10', 'L/10'] 
    edge_labels = {(l1,l2):e_labels[1], (l2,l3):e_labels[3], (l3,l4):e_labels[4], (l3,l5):e_labels[5]}

    pos = {l1: (10, 40), l2: (100, 40), l3:(210, 40), l4:(290, 60), l5:(290, 20)}

    nx.draw_networkx_nodes(G, pos=pos, nodelist=list(pos.keys()), node_size=2000, alpha=1, label_width=100, ax=ax)
    nx.draw_networkx_edges(G, pos=pos, edgelist= edgelist , edge_style= 'solid', edge_color="gray", arrows=True, arrowsize=10, arrowstyle='wedge', ax=ax)

    nx.draw_networkx_edge_labels(G,pos,edge_labels=edge_labels, font_size=15, ax=ax);

    nx.draw_networkx_labels(G, pos=pos, labels=dict(zip(pos.keys(),pos.keys())),  font_color="black", ax=ax)

    # draw a and b
    plt.text(10, 40+3, s= e_labels[0] , fontsize=16, bbox=dict(facecolor='yellow', alpha=0.0),horizontalalignment='center')
    plt.text(100, 40+3, s= e_labels[2] , fontsize=16, bbox=dict(facecolor='yellow', alpha=0.0),horizontalalignment='center')


    popul = pop
    proba = [1, b, d, e, f]
    xx = list(pos.keys())

    # N
    x,y= pos[xx[0]]
    popul_N = int(popul*proba[0])
    plt.text(x,y-3.2,s= popul_N, fontsize=10, bbox=dict(facecolor='yellow', alpha=0.6),horizontalalignment='center')

    # S
    x,y= pos[xx[1]]
    popul_S = int(popul_N*proba[1])
    plt.text(x,y-3.2,s= popul_S, fontsize=10, bbox=dict(facecolor='yellow', alpha=0.6),horizontalalignment='center')

    # I
    x,y= pos[xx[2]]
    popul_I = int(popul_S*proba[2])
    plt.text(x,y-3.2,s= popul_I, fontsize=10, bbox=dict(facecolor='yellow', alpha=0.6),horizontalalignment='center')
    
    # U
    x,y= pos[xx[3]]
    popul_U = int(popul_I*proba[3])
    plt.text(x,y-3.2,s= popul_U, fontsize=10, bbox=dict(facecolor='yellow', alpha=0.6),horizontalalignment='center')

    # K
    x,y= pos[xx[4]]
    popul_K = int(popul_I*proba[4])
    plt.text(x,y-3.2,s= popul_K, fontsize=10, bbox=dict(facecolor='yellow', alpha=0.6),horizontalalignment='center')


    # R
    x,y= 340, 30
    popul_R = 'R= '+ str(int(popul_I * (1-death_rate)))
    plt.text(x,y-3.2,s= popul_R, fontsize=18, bbox=dict(facecolor='yellow', alpha=0.6),horizontalalignment='center')

    # D
    x,y= 340, 45
    popul_D = 'D= '+ str(int(popul_I * death_rate))
    plt.text(x,y-3.2,s= popul_D, fontsize=18, bbox=dict(facecolor='yellow', alpha=0.6),horizontalalignment='center')


    # NewI
    x,y= 340, 60
    popul_NewI = 'NewI= ' + str(int(((popul_U*14)+(popul_K*7))*(d+b)))
    plt.text(x,y-3.2,s= popul_NewI, fontsize=18, bbox=dict(facecolor='yellow', alpha=0.6),horizontalalignment='center')


    week_pop = [popul_N, popul_S, popul_I, popul_U, popul_K, popul_R, popul_D, popul_NewI]
    
    plt.title('h_score= '+str(h_score)+' | e_score= '+str(e_score), ha="center", va="bottom", size=20, color="red", fontweight="bold")
               
    ax.set_xticks([])
    ax.set_yticks([])

    plt.show()

Legend¶

The SIR Model:¶

            N : Normal, S : Susceptible, I : Infected
            U : Uknown carriers,  K : Known carriers
            D : Deaths,  R : Recovered,  NewI : New infections
            h_score : health score,  e_score: economic_score

The Reinforcement Learning components:¶

     States are intensities of : contact rate, connection rate, Identif rate, Death rate, health score, economic score
     Actions are intensities of : Travel restriction, Lock_down, Barriers, Social_dist, Testing rate, HC capacity
     Reward is a basic combination of : health score and economic score
