import networkx as nx   
from ipywidgets import interactive, HBox, Layout, VBox
from IPython.display import display
import ipywidgets as wg
import matplotlib.pyplot as plt

global G

def showNet(popul, measures):
    global G
    
    if measures == '1-Nothing':
        zz = 1
    if measures == '2-quarantine':
        zz = 2
    if measures == '3-lock-down':
        zz = 3
    if measures == '4-Test and isolate':
        zz = 4
    if measures == '2, 3 and 4':
        zz = 5
        
        
    fig, ax = plt.subplots(figsize=(15,10))
    G = nx.DiGraph()

    # Calculate for dynamic label changes
    l1 = zz
    
    c1, c2= zz , zz*2 
    
    l1, l2, l3, l4, l5, l6, l7 ='N', 'S', 'I', 'U', 'K', 'R', 'D'
    
    edgelist= [(l1,l1),(l1,l2),(l2,l3),   (l3,l4), (l3,l5),    (l4,l5), (l4,l6), (l4,l7),   (l5,l6),(l5,l7)]
    
    
    e_labels = ['a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k'] 
    edge_labels = {(l1,l1):e_labels[0], (l1,l2):e_labels[1],  (l2,l2):e_labels[2],(l2,l3):e_labels[3],   
                   (l3,l4):e_labels[4], (l3,l5):e_labels[5],  (l4,l5):e_labels[6], (l4,l6):e_labels[7], (l4,l7):e_labels[8],   
                   (l5,l6):e_labels[9],(l5,l7):e_labels[10]}
    
    pos = {l1: (10, 40), l2: (100, 40), l3:(210, 40), l4:(280, 60), l5:(280, 20), l6:(400, 20), l7:(380, 60)}
    
    nx.draw_networkx_nodes(G, pos=pos, nodelist=list(pos.keys()), node_size=2000, alpha=1, label_width=100, ax=ax)
    nx.draw_networkx_edges(G, pos=pos, edgelist= edgelist , edge_style= 'solid', edge_color="gray", arrows=True, arrowsize=10, arrowstyle='wedge', ax=ax)
    
    nx.draw_networkx_edge_labels(G,pos,edge_labels=edge_labels, font_size=15, ax=ax);
    nx.draw_networkx_labels(G, pos=pos, labels=dict(zip(pos.keys(),pos.keys())),  font_color="black", ax=ax)
    plt.text(10, 40+3, s= e_labels[0] , fontsize=16, bbox=dict(facecolor='yellow', alpha=0.0),horizontalalignment='center')
    plt.text(100, 40+3, s= e_labels[2] , fontsize=16, bbox=dict(facecolor='yellow', alpha=0.0),horizontalalignment='center')
    
    xx, yy = 100, 60
    new_i = 'U*(1-i-h-g)+K*(1-j-k) \n + I coming from other nodes' # (e.g. travelers)'
    plt.text(xx, yy ,s= 'NewInfections: '+str(new_i) , fontsize=16, bbox=dict(facecolor='white', alpha=0.6),horizontalalignment='center')
    plt.plot([100, 100], [59, 45], 'gray', linestyle='-', lw=1)
    plt.scatter(100, 45, marker='o', c='black')
    
    j = int(popul)
    for i in pos.keys():
        x,y= pos[i]
        plt.text(x,y-3.2,s= j, fontsize=10, bbox=dict(facecolor='yellow', alpha=0.6),horizontalalignment='center')
        j = int(j-(j*0.19*c1))
        
#     plt.title('week: 1')
    ax.set_xticks([])
    ax.set_yticks([])
    
inp = interactive(showNet, popul=(100,100000), measures=['1-Nothing','2-quarantine','3-lock-down','4-Test and isolate','2, 3 and 4']);
display(inp)
